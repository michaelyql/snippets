# CF491C - Deciphering

## Summary

- Given two strings $c$ and $a$ made using an alphabet of $k$ characters, find a 1-to-1 mapping of the characters in $c$ to the alphabet such that the number of matches $c[i]==a[i]$ is maximized.
- Constraints: 
    - $1\leq n\leq 2\cdot 10^6$
    - $1\leq k\leq 52$

[Link to Problem Statement](https://codeforces.com/problemset/problem/491/C)

## Solution

- We can use min-cost max-flow and find the "minimum cost" to push exactly $k$ units of flow through a flow graph, i.e. a complete matching to ensure that every alphabet is matched
- The size of the alphabet is small (at most $52\times 52$), so MCMF should pass comfortably
- Construct a graph with the vertices $0$ to $k-1$ on the left side representing the characters in $c$, and $k$ to $2k-1$ on the right side representing the characters being mapped onto
- Add edges from every vertex on the left to every vertex on the right 
    - The cost of an edge from $i$ to $j$ is equal to the number of matches where character $i$ in $c$ is matched to character $j$ in $a$
- Since MCMF minimizes the cost, we invert the number of matches (i.e. negate the cost), so when the cost is minimized, the number of matches is maximized

```cpp
struct MCMF 
{
    struct Edge
    {
        int from, to, capacity, cost;
        Edge(int from_, int to_, int cap_, int cost_):
            from(from_), to(to_), capacity(cap_), cost(cost_) {}
    };

    vector<vector<int>> adj, cost, capacity;
    vector<Edge> edges;

    const int INF = 1e9;

    void shortest_paths(int n, int v0, vector<int>& d, vector<int>& p) 
    {
        d.assign(n, INF);
        d[v0] = 0;
        vector<bool> inq(n, false);
        queue<int> q;
        q.push(v0);
        p.assign(n, -1);

        while (!q.empty()) 
        {
            int u = q.front();
            q.pop();
            inq[u] = false;
            for (int v : adj[u]) 
            {
                if (capacity[u][v] > 0 && d[v] > d[u] + cost[u][v]) 
                {
                    d[v] = d[u] + cost[u][v];
                    p[v] = u;
                    if (!inq[v]) 
                    {
                        inq[v] = true;
                        q.push(v);
                    }
                }
            }
        }
    }

    // Find min cost to achieve a flow of K exactly
    // Set K to INF to compute min-cost max-flow
    // Returns -1 if flow of exactly K is not achievable
    int min_cost_flow(int N, vector<Edge> edges, int K, int s, int t) 
    {
        adj.assign(N, vector<int>());
        cost.assign(N, vector<int>(N, 0));
        capacity.assign(N, vector<int>(N, 0));
        for (Edge e : edges) 
        {
            adj[e.from].push_back(e.to);
            adj[e.to].push_back(e.from);
            cost[e.from][e.to] = e.cost;
            cost[e.to][e.from] = -e.cost;
            capacity[e.from][e.to] = e.capacity;
        }

        int flow = 0;
        int total_cost = 0;
        vector<int> d, p;
        // Successively shortest paths algorithm
        while (flow < K) 
        {
            shortest_paths(N, s, d, p);
            if (d[t] == INF)
                break;

            // find max flow on that path
            int f = K - flow;
            int cur = t;
            while (cur != s) 
            {
                f = min(f, capacity[p[cur]][cur]);
                cur = p[cur];
            }

            // apply flow
            flow += f;
            total_cost += f * d[t]; // cost is flow * cost per unit flow to t
            cur = t;
            while (cur != s) 
            {
                capacity[p[cur]][cur] -= f;
                capacity[cur][p[cur]] += f;
                cur = p[cur];
            }
        }

        if (flow < K)
            return -1;
        else
            return total_cost;
    }
};

int val(char c) {
    if (islower(c)) return c - 'a';
    else return c - 'A' + 26;
}

char to_char(int v) {
    if (v < 26) return v + 'a';
    else return v - 26 + 'A';
}

void solve(){
    int n, k;
    cin >> n >> k;
    string c, a;
    cin >> c >> a;
    MCMF mcmf;
    vvi mp(k, vi(k, 0));
    for(int i=0;i<n;i++){
        mp[val(c[i])][val(a[i])]++;
    }
    // left side: [0, k-1]
    // right side: [k, 2k-1]
    // source: 2k
    // sink: 2k+1

    vector<MCMF::Edge> edges;
    int src = 2 * k, sink = src + 1;

    for(int i=0;i<k;i++){
        edges.emplace_back(src, i, 1, 0);
        for(int j=0;j<k;j++){
            // assign char i to j
            edges.emplace_back(i, j+k, 1, -mp[i][j]);
        }
        edges.emplace_back(i+k, sink, 1, 0);
    }
    int f = mcmf.min_cost_flow(2*k+2, edges, k, src, sink);
    cout << -f << '\n';

    // restore flows
    vi mapping(k, -1);
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
            if (mcmf.capacity[i][j+k] == 0) { 
                mapping[i] = j; 
                break; 
            }
        }
    }
    for (int x : mapping) cout << to_char(x);
}
```
